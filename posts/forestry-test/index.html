#!/bin/bash

export NIC=ens37
export IPADDR=`ip addr show $NIC | grep "inet " | sed 's/^[ \t]*//g' | sed 's/[ \t]\+/ /g' | cut -d' ' -f 2 | cut -d'/' -f 1`

## install depedencies
yum update && \
yum install git bind dhcp xinetd fence-agents debmirror syslinux && \
yum install $(repoquery --requires cobbler)

## enable tftpd
sed -i -e 's/= yes/= no/' /etc/xinetd.d/tftp

## build / install cobbler from source
git clone https://github.com/cobbler/cobbler.git
cd cobbler/
make
make install
cd ../
rm -rf cobbler
